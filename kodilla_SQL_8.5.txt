USE kodilla_tester;
select database();

SELECT I.SUMMARY, IL.NAME,I.DESCRIPTION
FROM ISSUES I
JOIN issueslists IL ON IL.ID = I.ISSUESLIST_ID
ORDER BY IL.NAME;

SELECT I.SUMMARY, IL.NAME,I.DESCRIPTION, U.LASTNAME, U.FIRSTNAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON IL.ID = I.ISSUESLIST_ID
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.LASTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS USER_TASKS
FROM ISSUES I
JOIN ISSUESLISTS IL ON IL.ID = I.ISSUESLIST_ID
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.FIRSTNAME, U.LASTNAME, I.USER_ID_ASSIGNEDTO;
